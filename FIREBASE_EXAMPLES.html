1. Register with OTP

Integration for register.html

<form id="registerForm">

         required>

         id="email" 
         placeholder="Email" 
         required>

         id="password" 
         placeholder="Password" 
         required>

  </button>
</form>

<div id="otpSection" style="display:none;">
  <input type="text" 
         id="otpCode" 
         placeholder="123456"

    Verify OTP
  </button>

JavaScript Implementation

// Send OTP for registration
async function sendRegisterOTP() {
  const phoneNumber = document.getElementById('phoneNumber').value;
  const email = document.getElementById('email').value;
  
  if (!phoneNumber || !email) {
    alert('Please enter phone and email');
    return;
  }
  
  try {
    const response = await fetch('/api/send-otp', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phoneNumber, email })
    });
    
    const data = await response.json();
    if (data.success) {
      document.getElementById('otpSection').style.display = 'block';
      document.getElementById('otpCode').focus();
    }
  } catch (error) {
    console.error('Error:', error);
  }
}

// Verify OTP for registration
async function verifyRegisterOTP() {
  const phoneNumber = document.getElementById('phoneNumber').value;
  const otp = document.getElementById('otpCode').value;
  
  try {
    const response = await fetch('/api/verify-otp', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phoneNumber, otp })
    });
    
    const data = await response.json();
    if (data.verified) {
      // Register user
      registerUser();
    } else {
      document.getElementById('otpMessage').textContent = 
        '❌ ' + data.message;
    }
  } catch (error) {
    console.error('Error:', error);
  }
}

// Save user to database after OTP verification
async function registerUser() {
  const userData = {
    full_name: document.getElementById('fullName').value,
    mobile: document.getElementById('phoneNumber').value,
    email: document.getElementById('email').value,
    password: document.getElementById('password').value,
    role: 'citizen'
  };
  
  try {
    const response = await fetch('/api/register', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(userData)
    });
    
    const data = await response.json();
    if (data.success) {
      alert('Registration successful!');
      window.location.href = '/login.html';
    }
  } catch (error) {
    console.error('Error:', error);
  }
}
Note: After OTP verification, save user to database with encrypted password.

// Send OTP for login
async function sendLoginOTP() {
  const phoneNumber = document.getElementById('otpPhone').value;
  
  if (!phoneNumber) {
    alert('Please enter phone number');
    return;
  }
  
  const response = await fetch('/api/send-otp', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ 
      phoneNumber, 
      email: phoneNumber + '@vconnect.local' 
    })
  });
  
  if (response.ok) {
    document.getElementById('otpVerifySection').style.display = 'block';
    document.getElementById('loginOtp').focus();
  }
}

// Verify OTP and login
async function verifyLoginOTP() {
  const phoneNumber = document.getElementById('otpPhone').value;
  const otp = document.getElementById('loginOtp').value;
  
  const verifyResponse = await fetch('/api/verify-otp', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ phoneNumber, otp })
  });
  
  const verifyData = await verifyResponse.json();
  if (verifyData.verified) {
    // Login user
    const loginResponse = await fetch('/api/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        mobile: phoneNumber,
        password: 'otp_authenticated' 
      })
    });
    
    const loginData = await loginResponse.json();
    if (loginData.success) {
      // Store user session
      sessionStorage.setItem('user', JSON.stringify({
        mobile: phoneNumber,
        role: loginData.role
      }));
      window.location.href = '/dashboard.html';
    }
  }
}
Security: Implement server-side session management for security.

// Request OTP for profile update
async function requestProfileUpdateOTP() {
  const phone = document.getElementById('phone').value;
  const email = document.getElementById('email').value;
  
  const response = await fetch('/api/send-profile-otp', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ 
      phoneNumber: phone, 
      email 
    })
  });
  
  if (response.ok) {
    document.getElementById('otpVerification').style.display = 'block';
    document.getElementById('profileOtp').focus();
  }
}

// Confirm and update profile
async function confirmProfileUpdate() {
  const otp = document.getElementById('profileOtp').value;
  const phone = document.getElementById('phone').value;
  
  // Verify OTP
  const verifyResponse = await fetch('/api/verify-otp', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ phoneNumber: phone, otp })
  });
  
  const verifyData = await verifyResponse.json();
  if (!verifyData.verified) {
    alert('Invalid OTP: ' + verifyData.message);
    return;
  }
  
  // Update profile
  const updateResponse = await fetch('/api/update-profile', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      mobile: phone,
      full_name: document.getElementById('fullName').value,
      village: document.getElementById('village').value,
      mandal: document.getElementById('mandal').value,
      district: document.getElementById('district').value,
      aadhaar: document.getElementById('aadhaar').value,
      profile_image: ''
    })
  });
  
  const updateData = await updateResponse.json();
  if (updateData.success) {
    alert('Profile updated successfully!');
    document.getElementById('otpVerification').style.display = 'none';
  }
}
Tip: Always verify OTP before making sensitive changes like profile updates.

  if (otpResendCooldown > 0) {
    alert(`Please wait ${otpResendCooldown}s before resending`);
    return;
  }

  await generateOTP();

  document.getElementById('resendBtn').disabled = true;
  
  const timer = setInterval(() => {
    otpResendCooldown--;
    document.getElementById('resendBtn').textContent = 
      `Resend in ${otpResendCooldown}s`;
    
    if (otpResendCooldown === 0) {

      document.getElementById('resendBtn').disabled = false;
      document.getElementById('resendBtn').textContent = 'Resend OTP';
    }
  }, 1000);
}

6. Error Handling

Proper error handling patterns

JavaScript Implementation

async function handleOTPWithErrorHandling() {
  try {
    // Validate input
    const phone = document.getElementById('phone').value;
    if (!validatePhoneNumber(phone)) {
      throw new Error('Invalid phone number format');
    }
    
    // Send OTP
    const response = await fetch('/api/send-otp', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phoneNumber: phone })
    });
    
    // Check response status
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const data = await response.json();
    
    if (data.success) {
      showSuccessMessage('OTP sent successfully');
    } else {
      throw new Error(data.error || 'Failed to send OTP');
    }
    
  } catch (error) {
    console.error('Error:', error);
    showErrorMessage(error.message);
  }
}

function validatePhoneNumber(phone) {
  // Validate +country code format
  return /^\+\d{1,3}\d{6,14}$/.test(phone);
}

function showErrorMessage(message) {
  const errorDiv = document.getElementById('errorMessage');
  errorDiv.textContent = '❌ ' + message;
  errorDiv.style.display = 'block';
  setTimeout(() => errorDiv.style.display = 'none', 5000);
}

function showSuccessMessage(message) {
  const successDiv = document.getElementById('successMessage');
  successDiv.textContent = '✅ ' + message;
  successDiv.style.display = 'block';
}
